vp-dev is a function
vp-dev () 
{ 
    if [ "$1" = "env" ] && [ "$2" = "use" ]; then
        case " $* " in 
            *" -h "* | *" --help "*)
                command vp-dev "$@";
                return
            ;;
        esac;
        __vp_out="$(command vp-dev "$@")" || return $?;
        eval "$__vp_out";
    else
        command vp-dev "$@";
    fi
}
